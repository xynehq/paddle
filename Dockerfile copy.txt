FROM nvidia/cuda:12.9.1-cudnn-runtime-ubuntu22.04

ENV DEBIAN_FRONTEND=noninteractive
ENV PADDLEX_HPS_DEVICE_TYPE=gpu

RUN apt-get update && apt-get install -y --no-install-recommends \
    python3.10 python3.10-distutils python3-pip git curl wget unzip && \
    ln -sf /usr/bin/python3.10 /usr/bin/python3 && \
    python3 -m pip install --upgrade pip && \
    rm -rf /var/lib/apt/lists/*

WORKDIR /app

# Install PaddlePaddle GPU built for CUDA 12.9
RUN pip install paddlepaddle-gpu==3.2.0 -i https://www.paddlepaddle.org.cn/packages/stable/cu129/

# Install PaddleX
RUN pip install paddlex==3.1.0

COPY . /app
WORKDIR /app/server

EXPOSE 8000 8001

CMD ["/bin/bash", "server.sh"]
